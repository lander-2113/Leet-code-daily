a bi-directional graph with n vertices,
meaning a undirected graph.

each vertex is labeled form 0 to n - 1.

graph rep: 2d int array.

task: valid path or  not from, source, to
vertex.

it is a simple graph traversal algorithm:
I am using detph first traversal.

code:
"""
need a adjacency list graph

then need a function that could be called recurisvely.

"""
class Solution:
    def validPath(self, n: int, edges: List[List[int]], source: int, destination: int) -> bool:
        if n < 2: return True
        graph = self.buildGraph(edges)
        print(graph)
        return self.hasPath(graph, source, destination, set())

    def hasPath(self, graph, src, dst, visited):
        if src == dst:
            return True
        if src in visited:
            return False
        visited.add(src)

        for n in graph.get(src):
            if self.hasPath(graph, n, dst, visited):
                return True
        return False

    def buildGraph(self, edges):
        graph = {}

        for edge in edges:

            a, b = edge[0], edge[1]

            if a not in graph:
                graph[a] = []
            if b not in graph:
                graph[b] = []

            graph[a].append(b)
            graph[b].append(a)

        return graph
