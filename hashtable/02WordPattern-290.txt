class Solution:
    def wordPattern(self, pattern: str, s: str) -> bool:        
        
        i = 0
        n = len(s)
        v = []
        temp = ''
        while i < len(s):
            if s[i] == ' ':
                v.append(temp)
                temp = ''
            else:
                temp += s[i]
            i += 1
        v.append(temp)
        
        if len(pattern) != len(v):
            return False
        
        visited = set()
        
        m = {}
        i = 0
        while i < len(pattern):
            if pattern[i] in m:
                if m[pattern[i]] != v[i]:
                    return False
            else:
                if v[i] in visited:
                    return False
                
                m[pattern[i]] = v[i]
                visited.add(v[i])
            i += 1
        return True
